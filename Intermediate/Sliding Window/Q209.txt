209. Minimum Size Subarray Sum

class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        int size = Integer.MAX_VALUE;
        int sum = 0;
        int j =0, i =0;
        while(j<nums.length){
            sum += nums[j];
            j++;
            while(sum >= target){
                size = Math.min(size, (j-i));
                sum -= nums[i];
                i++;
            }
        }
        return size==Integer.MAX_VALUE?0:size;
    }
}
