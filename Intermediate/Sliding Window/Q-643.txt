643. Maximum Average Subarray I

class Solution {
    public double findMaxAverage(int[] nums, int k) {
        double sum = 0;
        int i = 0;
        while(i<k){
            sum += nums[i];
            i++;
        }
        double newSum = sum;
        int j= 0;
        while(j+k<nums.length){
            sum-=nums[j];
            sum+=nums[j+k];
            j++;
            newSum = Math.max(sum,newSum);
        }        
        return newSum/k;
    }
}
