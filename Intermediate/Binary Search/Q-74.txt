74. Search a 2D Matrix

class Solution {
    public boolean searchMatrix(int[][] matrix, int target) {
        // first binary search on rows
        // then binary search the mid row we found from previous BS 
        int top = 0;
        int bottom = matrix.length-1;
        while(top <= bottom){
            int vertMid = top + (bottom-top)/2;
            int left =0;
            int right = matrix[vertMid].length-1;
            if(target >= matrix[vertMid][left] && target <= matrix[vertMid][right]){
                while(left<=right){
                    int mid = left + (right-left)/2;

                    if(matrix[vertMid][mid]==target){
                        return true;
                    }else if(matrix[vertMid][mid]>target){
                        right = mid-1;
                    }else{
                        left = mid+1;
                    }
                }
                return false;
            }else if(target < matrix[vertMid][left]){
                bottom = vertMid-1;
            }else{
                top = vertMid+1;
            }
        }
        return false;
    }
}
