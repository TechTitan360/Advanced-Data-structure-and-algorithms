1343. Number of Sub-arrays of Size K and Average Greater than or Equal to Threshold

class Solution {
    public int numOfSubarrays(int[] arr, int k, int threshold) {
        int sum = 0;
        int count = 0;
        for(int i = 0; i<k; i++){
            sum+=arr[i];
        }
        if(sum/k >= threshold ) count++;
        int i = 0;
        while(i+k<arr.length){
            sum-=arr[i];
            sum+=arr[i+k];
            i++;
            if(sum/k >= threshold ) count++;
        }
        return count;
    }
}
