1733. Minimum Number of People to Teach

class Solution {
    public int minimumTeachings(int n, int[][] languages, int[][] friendships) {
        int m = languages.length;
        
        List<Set<Integer>> langs = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            Set<Integer> set = new HashSet<>();
            for (int lang : languages[i]) {
                set.add(lang);
            }
            langs.add(set);
        }
        
        Set<Integer> brokenUsers = new HashSet<>();
        for (int[] f : friendships) {
            int u = f[0] - 1;
            int v = f[1] - 1;

            boolean common = false;
            for (int lang : langs.get(u)) {
                if (langs.get(v).contains(lang)) {
                    common = true;
                    break;
                }
            }

            if (!common) {
                brokenUsers.add(u);
                brokenUsers.add(v);
            }
        }
        

        if (brokenUsers.isEmpty()) return 0;
        

        int minTeach = Integer.MAX_VALUE;
        for (int lang = 1; lang <= n; lang++) {
            int needTeach = 0;
            for (int user : brokenUsers) {
                if (!langs.get(user).contains(lang)) {
                    needTeach++;
                }
            }
            minTeach = Math.min(minTeach, needTeach);
        }
        
        return minTeach;
    }
}
